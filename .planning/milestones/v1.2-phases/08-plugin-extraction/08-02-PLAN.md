```typescript
// workersPlugin needs to call eventBus.subscribe() — this is async.
// Pattern: async factory that accepts the already-constructed servicesPlugin instance,
// calls subscribe(), then returns the Elysia plugin.
// The plugin itself does NOT need to .use(servicesPlugin) if services are passed as args.

export async function createWorkersPlugin(
  eventBus: PgBossEventBus,
  notificationService: NotificationService,
  auditService: AuditService,
) {
  await eventBus.subscribe("user.registered", ..., "notification");
  await eventBus.subscribe("user.registered", ..., "audit");
  return new Elysia({ name: "workers" }); // no routes — purely side effects
}

// userRoutesPlugin: synchronous — services accessed from context (via .use(servicesPlugin))
export const userRoutesPlugin = new Elysia({ name: "user-routes" })
  .get("/users", ({ userRepo }) => userRepo.findAll())
  .post("/users", async ({ userService, body, set }) => { ... })
// For this to typecheck, servicesPlugin must be .use()'d BEFORE userRoutesPlugin
// OR userRoutesPlugin must explicitly .use(servicesPlugin) internally.
// PREFERRED for independence/testability: userRoutesPlugin internally .use(servicesPlugin).
// BUT since servicesPlugin is async-constructed (factory), route plugin cannot statically import it.
// CORRECT approach: userRoutesPlugin is a FUNCTION accepting the servicesPlugin instance:
export function createUserRoutesPlugin(services: Awaited<ReturnType<typeof createServicesPlugin>>) {
  return new Elysia({ name: "user-routes" })
    .use(services)
    .get("/users", ({ userRepo }) => userRepo.findAll())
    .post("/users", async ({ userService, body, set }) => { ... })
}
```

```typescript
/**
 * workersPlugin: Elysia plugin that registers all event bus subscriptions.
 *
 * Use createWorkersPlugin() — it is async because eventBus.subscribe() is async.
 * MUST be awaited before app.listen() — boot order: subscribe ALL workers before HTTP server starts.
 *
 * Encapsulates: all boss.subscribe() / boss.work() calls for domain event handlers.
 * Zero subscription wiring should live outside this plugin.
 */
import { Elysia } from "elysia";
import type { PgBossEventBus } from "../infrastructure/events/PgBossEventBus.ts";
import type { NotificationService } from "../domains/notification/NotificationService.ts";
import type { AuditService } from "../domains/audit/AuditService.ts";

export async function createWorkersPlugin(
  eventBus: PgBossEventBus,
  notificationService: NotificationService,
  auditService: AuditService,
) {
  // Subscribe ALL workers before server starts — boot order enforces fan-out correctness
  // Each subscribe() call: createQueue → boss.subscribe → boss.work
  // Boot order: start → (createQueue + subscribe + work) × N subscribers → listen
  await eventBus.subscribe(
    "user.registered",
    (payload) => notificationService.handleUserRegistered(payload),
    "notification",
  );
  console.log("[app] NotificationService subscribed to user.registered.");

  await eventBus.subscribe(
    "user.registered",
    (payload) => auditService.handleUserRegistered(payload),
    "audit",
  );
  console.log("[app] AuditService subscribed to user.registered.");

  return new Elysia({ name: "workers" });
}
```
