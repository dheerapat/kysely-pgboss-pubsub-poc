  // ... setup (unchanged) ...

  // 5. Subscribe ALL workers BEFORE server starts — fan-out boot sequence
  //    Boot order: start → createQueue → subscribe → work (for each subscriber) → listen

    "user.registered",
    (payload) => notificationService.handleUserRegistered(payload),
    "notification",  // → queue: "notification.user.registered"
  );

  console.log("[app] AuditService subscribed to user.registered.");  // NEW

  // THEN start server — all subscriptions registered before HTTP is accepted
  const app = new Elysia()
    .get("/users", ...)
