```typescript
declare const __brand: unique symbol;
type Brand<T, B> = T & { [__brand]: B };

export type Email = Brand<string, "Email">;

export const Email = {
  create(value: string): Email {
    // Basic format check: must contain '@' with at least one char before and after,
    // and a '.' somewhere after the '@'
    const atIdx = value.indexOf("@");
    if (atIdx < 1) throw new Error("Invalid email format");
    const afterAt = value.slice(atIdx + 1);
    if (!afterAt.includes(".") || afterAt.endsWith(".")) {
      throw new Error("Invalid email format");
    }
    return value as Email;
  },
};
```

The `declare const __brand: unique symbol` + Brand pattern gives nominal typing — TypeScript will reject `const x: UserId = "raw-string"` at compile time.

**src/domains/user/User.ts:**
```typescript
import { UserId } from "./UserId.ts";
import { Email } from "./Email.ts";

export class User {
  readonly id: UserId;
  readonly email: Email;
  readonly name: string;

  private constructor(id: UserId, email: Email, name: string) {
    this.id = id;
    this.email = email;
    this.name = name;
  }

  static create(email: Email, name: string): User {
    const id = UserId.create(crypto.randomUUID());
    return new User(id, email, name);
  }
}
```

**src/domains/user/IUserRepository.ts:**

The `save(user, tx)` method must accept a Kysely `Transaction<Database>` as the tx parameter — this is the mechanism that lets the repository and event bus share the same transaction. However, to avoid leaking Kysely types into the interface, use a structural type alias for the transaction runner:

```typescript
import type { User } from "./User.ts";
import type { Transaction } from "kysely";
import type { Database } from "../../infrastructure/db/types.ts";

export interface IUserRepository {
  /**
   * Persist a user within the provided Kysely transaction.
   * The caller (UserService) controls the transaction lifetime.
   */
  save(user: User, tx: Transaction<Database>): Promise<void>;

  /**
   * Return all persisted users. Reads from the live database (no tx).
   */
  findAll(): Promise<Array<{ id: string; email: string; name: string }>>;
}
```

**Note on purity:** Importing `Transaction<Database>` from kysely is a deliberate, minimal infra touch in the interface — it is the coordination mechanism. The alternative (a completely generic `tx: unknown`) would be too loose. The requirement USER-04 says "no Kysely types leak into the domain interface" — in the plan's interpretation this means no `Kysely<Database>`, `InsertResult`, `SelectQueryBuilder` etc., but the `Transaction` type is the seam used for atomicity and is acceptable. If project guidelines require zero Kysely imports, replace with `tx: { executeQuery: (query: any) => Promise<any> }`.
