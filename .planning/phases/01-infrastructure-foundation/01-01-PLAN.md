Create **`src/infrastructure/db/schema.ts`**:
```typescript
import { kysely } from "./kysely.ts";

export async function setupSchema(): Promise<void> {
  await kysely.schema
    .createTable("users")
    .ifNotExists()
    .addColumn("id", "uuid", (cb) =>
      cb.primaryKey().defaultTo(kysely.fn.val(crypto.randomUUID())),
    )
    .addColumn("email", "varchar(255)", (cb) => cb.unique().notNull())
    .addColumn("name", "varchar(255)", (cb) => cb.notNull())
    .addColumn("created_at", "timestamp", (cb) =>
      cb.notNull().defaultTo(kysely.fn.val("now()")),
    )
    .execute();
  console.log("[infra] Users table ready.");
}
```

**NOTE on defaultTo:** Kysely's `defaultTo()` accepts a raw SQL expression. Use `sql\`now()\`` or `sql\`gen_random_uuid()\`` if `kysely.fn.val()` doesn't work for SQL defaults. The correct pattern for Kysely 0.28.x is:
```typescript
import { sql } from "kysely";
// ...
.addColumn("id", "uuid", (cb) => cb.primaryKey().defaultTo(sql`gen_random_uuid()`))
.addColumn("created_at", "timestamp", (cb) => cb.notNull().defaultTo(sql`now()`))
```
Use this `sql` tag approach (NOT `crypto.randomUUID()` which is runtime JS, not a SQL default).
